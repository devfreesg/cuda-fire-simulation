00130   void BitmapWriter::flush(std::string fileName)
00131   {
00132     FILE *f = 0;
00133     fopen_s(&f, fileName.c_str(),"wb");
00134     if (f == 0)
00135       return;
00136     char magic[2]={'B','M'};
00137     fwrite(magic, 2, 1, f);
00138     fwrite((void*)&m_fileHeader, sizeof(BMPFileHeader), 1, f);
00139     fwrite((void*)&m_fileInfoHeader, sizeof(BMPInfoHeader), 1, f);
00140     fwrite(m_data, m_fileInfoHeader.biSizeImage, 1, f);
00141     fclose(f);
00142   }
